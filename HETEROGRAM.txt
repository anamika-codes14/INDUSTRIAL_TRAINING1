                                             HETEROGRAM:
                                        ==================
                                        
----->>>>> A word/phrase where NO letter of the alphabet appears more than once
___________________________________________________________________________________



#include <stdio.h>

#include <string.h>

#include <ctype.h>


// Returns 1 if heterogram (no letter repeats), 0 otherwise

int isHeterogram(char *str) 
{
    int freq[26] = {0};
    
    for (int i = 0; str[i]; i++)
    {
        if (isalpha(str[i]))
        {
            int idx = tolower(str[i]) - 'a';
            
            freq[idx]++;
            
            if (freq[idx] > 1) return 0;
            
        }
    }
    return 1;
    
}

// Count distinct letters used

int distinctLetters(char *str) 
{
    int seen[26] = {0};
    
    int count = 0;
    
    for (int i = 0; str[i]; i++)
    
        if (isalpha(str[i]) && !seen[tolower(str[i]) - 'a'])
        
            seen[tolower(str[i]) - 'a'] = 1, count++;
            
    return count;
    
}

int main() 
{
    char str[200];
    
    printf("Enter a word/phrase: ");
    
    fgets(str, sizeof(str), stdin);
    
    str[strcspn(str, "\n")] = '\0';
    

    if (isHeterogram(str))
    
        printf("\"%s\" IS a heterogram! (%d distinct letters, none repeated)\n",
        
               str, distinctLetters(str));
               
    else
        printf("\"%s\" is NOT a heterogram. (Some letters repeat)\n", str);
        

    return 0;
    
}
