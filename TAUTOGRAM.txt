                                               TAUTOGRAM: 
                                            =================
                                            
-------->>>>> All words in a phrase start with the same letter
______________________________________________________________________


// Example: "Silly Sally sells seashells" — all start with 'S'
#include <stdio.h>

#include <string.h>

#include <ctype.h>


int isTautogram(char *str) 
{
    char firstLetter = '\0';
    

    // Find first alphabetic character (start of first word)
    
    int i = 0;
    
    while (str[i] && !isalpha(str[i])) i++;
    
    if (!str[i]) return 0; // No letters at all
    

    firstLetter = tolower(str[i]);

    // Now check every word's first letter
    
    int inWord = 0;
    
    
    for (; str[i]; i++) {
    
        if (isalpha(str[i]))
        {
            if (!inWord) 
            {
                // Start of a new word
                
                if (tolower(str[i]) != firstLetter)
                
                    return 0;
                    
                inWord = 1;
            }
        } else 
        {
            inWord = 0; // Space or punctuation resets word boundary
        }
    }
    return 1;
}

int main()
{
    char str[200];
    
    printf("Enter a phrase: ");
    
    fgets(str, sizeof(str), stdin);
    
    str[strcspn(str, "\n")] = '\0';

    if (isTautogram(str))
    
        printf("\"%s\" IS a tautogram! (All words start with the same letter)\n", str);
    else
        printf("\"%s\" is NOT a tautogram.\n", str);

    return 0;
}
