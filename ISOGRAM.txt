//                                                          ISOGRAM: 
                                                         ==============
                                                         
------->>>>> A word where no letter repeats .
             __________________________________



#include <stdio.h>

#include <string.h>

#include <ctype.h>


int isIsogram(char *str)
{
    int freq[26] = {0};
    
    for (int i = 0; str[i]; i++) 
    {
        if (isalpha(str[i])) 
        {
            int idx = tolower(str[i]) - 'a';
            
            if (freq[idx]++) return 0; // Letter repeated
            
        }
    }
    
    return 1;
    
}

int main() 
{
    char str[100];
    
    printf("Enter a word: ");
    
    fgets(str, sizeof(str), stdin);
    
    str[strcspn(str, "\n")] = '\0';
    

    if (isIsogram(str))
    
        printf("\"%s\" IS an isogram. (No letter repeats)\n", str);
    else
        printf("\"%s\" is NOT an isogram.\n", str);

    return 0;
}
