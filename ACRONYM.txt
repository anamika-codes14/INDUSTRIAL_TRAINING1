#include <stdio.h>

#include <string.h>

#include <ctype.h>

// Small words to skip (optional — comment out the check to include all)
const char *skipWords[] = {"and", "or", "of", "the", "a", "an", "in", "on", "at", "to", NULL};

int shouldSkip(char *word)
{
    char lower[50];
    int i;
    for (i = 0; word[i] && i < 49; i++)
    
        lower[i] = tolower(word[i]);
        
    lower[i] = '\0';

    for (int j = 0; skipWords[j]; j++)
    
        if (strcmp(lower, skipWords[j]) == 0) return 1;
    return 0;
}

void generateAcronym(char *str, char *acronym) 
{
    char temp[200];
    
    strncpy(temp, str, sizeof(temp) - 1);
    
    temp[sizeof(temp)-1] = '\0';

    int idx = 0;
    char *token = strtok(temp, " \t\n");
    
    while (token) {
        if (!shouldSkip(token) && isalpha(token[0]))
        
            acronym[idx++] = toupper(token[0]);
            
        token = strtok(NULL, " \t\n");
    }
    acronym[idx] = '\0';
}

int main() {
    char str[200];
    
    char acronym[100];

    printf("Enter a phrase: ");
    
    fgets(str, sizeof(str), stdin);
    
    str[strcspn(str, "\n")] = '\0';

    generateAcronym(str, acronym);

    printf("Phrase  : %s\n", str);
    
    printf("Acronym : %s\n", acronym);

    return 0;
}
